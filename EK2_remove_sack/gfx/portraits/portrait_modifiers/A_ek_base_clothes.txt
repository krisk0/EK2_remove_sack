clothes_effect = { # Gene modifers to apply for characters wearing or not wearing clothes
	usage = game
	selection_behavior = max
	fallback = unclothed_body
	priority = 20

	clothed_body = {
		weight = {
			base = 0
			modifier = {
				add = 1000
				NOR = {
					should_be_fully_naked_portrait_trigger = yes
					should_be_topless_portrait_trigger = yes
				}
			}  
		}    
	}

	unclothed_body = {
		dna_modifiers = {          
		}
		weight = {
			base = 10
		}
	}
	
	unclothed_body_event = {
		dna_modifiers = {          
		}
		outfit_tags = { no_clothes }        
		require_outfit_tags = yes
		weight = {
			base = 1000
		}
	}
}

clothes = { # Special group for the ruler designer only. The game expects this to be named "clothes"
	usage = game
	selection_behavior = weighted_random
	priority = 20

	ruler_designer_clothes = {
		dna_modifiers = {
			accessory = {
				mode = add
				gene = clothes
				template = no_clothes
				range = { 0 1 }
			}
		}
		weight = {
			base = 0
			modifier = {
				add = 100
				portrait_ruler_designer_trigger = yes
				# Removing `exits = this` below results in multiple error messages.
				#  Removing both checks `exists...` and `age...` results in fig leaf
				#  covering genitals
				exists = this
				age >= 11
			}
		}
	}

	ruler_designer_clothes_child = {
		dna_modifiers = {
			accessory = {
				mode = add
				gene = clothes
				template = western_bedchamber
				range = { 0 1 }
			}
		}
		weight = {
			base = 0
			modifier = {
				add = 50
				portrait_ruler_designer_trigger = yes
			}
		}
	}
}
